# YouTube API 연동을 위한 환경변수 설정 안내
# 이 내용을 .env 파일에 추가하세요

# ===== YouTube API 설정 =====
# Google Cloud Console에서 YouTube Data API v3 키를 발급받아 설정하세요
# 1. Google Cloud Console (https://console.cloud.google.com) 접속
# 2. 새 프로젝트 생성 또는 기존 프로젝트 선택
# 3. "API 및 서비스" > "라이브러리"에서 "YouTube Data API v3" 활성화
# 4. "API 및 서비스" > "사용자 인증 정보"에서 API 키 생성
# 5. API 키에 YouTube Data API v3 제한 설정 (보안)

YOUTUBE_API_KEY=your_youtube_api_key_here

# ===== 기존 설정 유지 =====
# 기존 .env 파일의 다른 설정들은 그대로 유지하고
# 위의 YOUTUBE_API_KEY만 추가하세요

# 예시:
# OPENAI_API_KEY=your_openai_api_key
# MONGODB_URI=mongodb://username:password@host:port/database
# MONGODB_DB_NAME=rag_database
# YOUTUBE_API_KEY=AIzaSyABC123DEF456GHI789JKL012MNO345PQR

# ===== YouTube API 할당량 정보 =====
# 무료 할당량: 하루 10,000 단위
# 검색 1회: 100 단위 소모
# 동영상 상세 정보 조회 1회: 1 단위 소모
# 따라서 하루 약 100회 정도의 검색이 가능합니다

# ===== 주의사항 =====
# 1. API 키는 절대 공개하지 마세요
# 2. .env 파일을 git에 업로드하지 마세요 (.gitignore에 추가됨)
# 3. API 키에 HTTP 리퍼러 제한을 설정하여 보안을 강화하세요
